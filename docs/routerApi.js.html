// @flow

import fetch from 'cross-fetch';
// import CONSTANTS from './../constants/constants';
/**
 * 
 * Router APIs
 * @class
 * 
 * @since 1.0.0
 */
const RouterAPI = {
    /**
     * 
     * start of the file.
     * @method
     * 
     * @since 1.0.0
     * @param {string} a Name of the person
     * @returns {string} Returns string
     * @example 
     * fun("Hello")
     * In console: Start... Hello
     */
    fun: (a: string)=> {
        console.log(`Start... ${a}`);
        return `Start... ${a}`
    },

    /**
     * 
     * Ping Router
     * @method
     * 
     * @since 1.0.0
     * @param {string} routerIp IP of the router
     * @returns {Object} Returns API response
     */
    ping: (routerIp:string) => {
        return fetch(`https://${routerIp}:8081/ping`,{
            method: 'PUT',
            headers: {
                'Accept': 'application/json',
                'Content-Type': 'application/json',
            },
        });
    },
    /**
     * 
     * Activation
     * @method
     * 
     * @since 1.0.0
     * @param {string} routerIp IP of the router
     * @returns {Object} Returns API response
     */
    activate: (routerIp:string) => {
        // const proxyurl = "https://cors-anywhere.herokuapp.com/";
        const url = `https://${routerIp}:8081/activation`;
        return fetch(url,{
            method: 'GET',
            headers: {
                'Accept': 'application/json',
                'Content-Type': 'application/json'
            }
        }).then((response) => {
            // console.log(response);
            if (response.status === 200) {
                return response.json();
            } else {
                return response;
            }
        });
    }
}

export default RouterAPI;
